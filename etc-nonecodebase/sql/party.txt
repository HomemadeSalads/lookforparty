  `id` INT NOT NULL AUTO_INCREMENT,
  `username` VARCHAR(100) NOT NULL,

  `reward` INT NULL,
  PRIMARY KEY (`postid`),
  CONSTRAINT `userid`
    FOREIGN KEY (`id`)
    REFERENCES `party`.`player` (`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE);
